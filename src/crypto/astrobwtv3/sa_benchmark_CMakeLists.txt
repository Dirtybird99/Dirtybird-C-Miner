# SA Construction Benchmark - CMake Build Configuration
#
# Usage:
#   mkdir build && cd build
#   cmake -DCMAKE_BUILD_TYPE=Release ..
#   cmake --build .
#
# Or with Ninja:
#   cmake -G Ninja -DCMAKE_BUILD_TYPE=Release ..
#   ninja

cmake_minimum_required(VERSION 3.16)
project(sa_benchmark CXX C)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_C_STANDARD 11)

# Optimization flags
set(CMAKE_C_FLAGS_RELEASE "-O3 -DNDEBUG")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -DNDEBUG")

# Architecture-specific flags
if(CMAKE_SYSTEM_PROCESSOR MATCHES "x86_64|AMD64")
    set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -march=native -mtune=native")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -march=native -mtune=native")
elseif(CMAKE_SYSTEM_PROCESSOR MATCHES "aarch64|arm64")
    set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -march=armv8-a+aes")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -march=armv8-a+aes")
endif()

# Include directories
include_directories(
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/../../../include
)

# Source files
set(BENCHMARK_SOURCES
    sa_benchmark.cpp
    sa_incremental.cpp

    # divsufsort library (C files)
    divsufsort.c
    sssort.c
    trsort.c
    utils.c
)

# Create executable
add_executable(sa_benchmark ${BENCHMARK_SOURCES})

# Compiler-specific flags
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    target_compile_options(sa_benchmark PRIVATE
        -Wall
        -Wextra
        -Wpedantic
        -fno-omit-frame-pointer  # Better profiling support
    )
endif()

if(MSVC)
    target_compile_options(sa_benchmark PRIVATE
        /W4
        /O2
        /arch:AVX2
    )
endif()

# Platform-specific linking
if(WIN32)
    target_link_libraries(sa_benchmark PRIVATE winpthread)
elseif(UNIX AND NOT APPLE)
    target_link_libraries(sa_benchmark PRIVATE pthread)
endif()

# Enable LTO for release builds
include(CheckIPOSupported)
check_ipo_supported(RESULT ipo_supported OUTPUT ipo_error)
if(ipo_supported)
    set_property(TARGET sa_benchmark PROPERTY INTERPROCEDURAL_OPTIMIZATION_RELEASE TRUE)
endif()

# Custom target to run benchmarks
add_custom_target(run_benchmark
    COMMAND sa_benchmark -n 10000 -s 256 -p astrobwt
    DEPENDS sa_benchmark
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    COMMENT "Running SA benchmark..."
)

# Custom target for quick test
add_custom_target(run_test
    COMMAND sa_benchmark -t
    DEPENDS sa_benchmark
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    COMMENT "Running SA correctness tests..."
)

# Custom target for incremental benchmark
add_custom_target(run_incremental
    COMMAND sa_benchmark -i -n 10000 -s 256
    DEPENDS sa_benchmark
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    COMMENT "Running incremental SA benchmark..."
)
